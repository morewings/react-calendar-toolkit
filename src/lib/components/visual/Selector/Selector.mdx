import { Playground, Props } from 'docz';
import Datepicker from 'lib';
import Selector from './Selector';

`Selector` component is rendered to the top of calendar. It allows changing [calendar precision](/#calendar-views-and-precision) (`setPrecision`) and stepping months (`incrementMonth`, `decrementMonth`). Replace it with custom component.

<Playground>
  {() => {
    const style = {
      color: 'black',
      textAlign: 'center',
      padding: '16px 0',
      border: '2px dashed pink',
      margin: '2px'
    };
    const CustomComponent = () => {
      return (
        <div style={style}>
          {`<Selector />`}
        </div>
      )
    };
    return (
      <Datepicker
        onDateSet={date => {console.log('date set', date);}}
        minPrecision="month"
        renderSelectorAs={CustomComponent}
        title="Demo datepicker" />
    )
  }}
</Playground>
