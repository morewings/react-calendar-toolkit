import { Playground, Props } from 'docz';
import Datepicker from 'lib';
import Day from './Day';

`Day` component displays specific date with [__day precision__](/#calendar-views-and-precision). `onDateSet` callback prop allows to set provided Date object as selected in `Datepicker`. Replace it with custom component.

<Playground>
  {() => {
    const style = {
      boxSizing: 'border-box',
      flex: '1 1 calc(100%/7 - 4px)',
      color: 'black',
      textAlign: 'center',
      padding: '0',
      fontSize: '12px',
      lineHeight: '30px',
      border: '2px dashed pink',
      margin: '2px'
    };
    const CustomComponent = ({
     onDateSet,
     date,
     isSameMonth,
     isToday,
     isSelected,
     isDisabled,
     isHighlighted,
     isWeekend,
     name
   }) => {
      const handleClick = () => {
        console.log('date set', date);
        onDateSet(date);
      };
      return (
        <div onClick={handleClick} style={style}>
          {name.numeric}
        </div>
      )
    };
    return (
      <Datepicker
        onDateSet={date => {console.log('date set', date);}}
        minPrecision="day"
        renderDayAs={CustomComponent}
        title="Demo datepicker" />
    )
  }}
</Playground>

## Wrap with custom component

You can provide wrapper (React Component or DOM node) to the day calendar.

<Playground>
  {() => {
    const style = {
      display: 'flex',
      width: 'auto',
      flexDirection: 'row',
      flexWrap: 'wrap',
      padding: '0 8px 8px',
      margin: '0 8px 8px',
      height: '247px',
      border: '2px dashed pink',
    };
    const CustomComponent = ({children}) => (
      <div style={style}>{children}</div>
    );
    return (
      <Datepicker
        onDateSet={date => {console.log('date set', date);}}
        minPrecision="day"
        wrapDaysWith={CustomComponent}
        title="Demo datepicker" />
    )
  }}
</Playground>
