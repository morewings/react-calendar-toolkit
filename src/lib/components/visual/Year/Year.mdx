import { Playground, Props } from 'docz';
import Datepicker from 'lib';
import Year from './Year';

`Year` component displays specific date with [__year precision__](/#calendar-views-and-precision). `onDateSet` callback prop allows to set provided Date object as selected in `Datepicker`. Replace it with custom component.

<Playground>
  {() => {
    const style = {
      boxSizing: 'border-box',
      flex: '0 1 calc(100% / 5 - 4px)',
      color: 'black',
      textAlign: 'center',
      padding: '0',
      border: '2px dashed pink',
      margin: '6px 2px',
      height: '36px',
      lineHeight: '36px'
    };
    const CustomComponent = ({
     isHighlighted,
     onDateSet,
     date,
     isSelected,
     isSameYear,
     name,
   }) => {
      const handleClick = () => {
        console.log('date set', date);
        onDateSet(date);
      };
      return (
        <div onClick={handleClick} style={style}>
          {name.numeric}
        </div>
      )
    };
    return (
      <Datepicker
        onDateSet={date => {console.log('date set', date);}}
        renderYearAs={CustomComponent}
        minPrecision="year"
        startDate={new Date(1999, 0, 1)}
        endDate={new Date(2020, 1, 25)}
        title="Demo datepicker" />
    )
  }}
</Playground>

## Wrap with custom CSS class

You can override CSS classname of Day Calendar element, see `yearCalendarClassName` prop.

```jsx
<Datepicker
  onDateSet={date => {console.log('date set', date);}}
  minPrecision="day"
  monthCalendarClassName="yearCalendarClassName"
  title="Demo datepicker" />
```
